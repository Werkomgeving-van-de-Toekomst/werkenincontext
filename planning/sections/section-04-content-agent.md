## Overview

The Content Generation Agent is responsible for generating document content using Markdown templates and context from the Research Agent. It fills template variables from multiple sources (user input, knowledge graph, agent-generated content) and produces the initial Markdown document that will be validated by the Compliance Agent.

The agent handles:
1. Loading the appropriate template for the document type
2. Resolving variables from multiple sources with priority ordering
3. Generating domain-specific content sections
4. Handling conditional template sections
5. Linking entities from the knowledge graph
6. Producing valid Markdown output

/// Execute the Content Generation Agent
/// 
/// # Arguments
/// * `request` - Original document request
/// * `research` - Research context from Research Agent
/// * `template` - Template to use for generation
/// * `kg_client` - Knowledge graph client for entity resolution
/// * `config` - Content generation configuration
/// 
/// # Returns
/// Generated document with Markdown content
pub async fn execute_content_agent(
    request: &DocumentRequest,
    research: &ResearchContext,
    template: &Template,
    kg_client: &GraphRagClient,
    config: &ContentConfig,
) -> Result<GeneratedDocument, ContentAgentError> {
    // Step 1: Resolve all template variables from available sources
    let variables = resolve_variables(
        request,
        research,
        template,
        kg_client,
        config,
    ).await?;
    
    // Step 2: Build template context for rendering
    let mut context = build_template_context(&variables);
    
    // Step 3: Add conditional section flags
    add_conditional_sections(&mut context, research, template);
    
    // Step 4: Generate AI-powered content sections
    generate_ai_sections(&mut context, research, config).await?;
    
    // Step 5: Render the template
    let engine = TemplateEngine::new()?;
    let content = engine.render_template(&template.content, &context)
        .map_err(ContentAgentError::TemplateRenderingError)?;
    
    // Step 6: Extract entity links
    let entity_links = if config.enable_entity_linking {
        extract_entity_links(&content, kg_client).await?
    } else {
        Vec::new()
    };
    
    // Step 7: Build section metadata
    let sections = extract_section_metadata(&content, template);
    
    Ok(GeneratedDocument {
        document_id: request.id,
        content,
        variables,
        entity_links,
        sections,
        generated_at: chrono::Utc::now(),
    })
}

/// Resolve template variables from multiple sources
/// 
/// Variable source priority:
/// 1. UserInput (direct from request)
/// 2. KnowledgeGraph (from entity queries)
/// 3. AgentGenerated (AI-generated content)
/// 4. Default (template default values)
async fn resolve_variables(
    request: &DocumentRequest,
    research: &ResearchContext,
    template: &Template,
    kg_client: &GraphRagClient,
    config: &ContentConfig,
) -> Result<Vec<TemplateVariable>, ContentAgentError> {
    let mut resolved_vars = Vec::new();
    
    // Track which variables have been resolved
    let mut resolved_names = HashMap::new();
    
    // Priority 1: User input (highest priority)
    for (key, value) in &request.context {
        resolved_names.insert(key.clone(), true);
        resolved_vars.push(TemplateVariable {
            name: key.clone(),
            value: value.clone(),
            source: VariableSource::UserInput,
        });
    }
    
    // Priority 2: Knowledge Graph
    for var_name in &template.required_variables {
        if !resolved_names.contains_key(var_name) {
            if let Some(value) = resolve_from_knowledge_graph(
                var_name,
                research,
                kg_client,
            ).await? {
                resolved_names.insert(var_name.clone(), true);
                resolved_vars.push(TemplateVariable {
                    name: var_name.clone(),
                    value,
                    source: VariableSource::KnowledgeGraph,
                });
            }
        }
    }
    
    // Priority 3: Agent-generated content (for remaining variables)
    for var_name in &template.required_variables {
        if !resolved_names.contains_key(var_name) {
            let value = generate_variable_value(
                var_name,
                request,
                research,
                config,
            ).await?;
            resolved_names.insert(var_name.clone(), true);
            resolved_vars.push(TemplateVariable {
                name: var_name.clone(),
                value,
                source: VariableSource::AgentGenerated,
            });
        }
    }
    
    // Verify all required variables are resolved
    for var_name in &template.required_variables {
        if !resolved_names.contains_key(var_name) {
            return Err(ContentAgentError::RequiredVariableMissing(
                var_name.clone()
            ));
        }
    }
    
    Ok(resolved_vars)
}

/// Resolve a variable value from the knowledge graph
async fn resolve_from_knowledge_graph(
    var_name: &str,
    research: &ResearchContext,
    kg_client: &GraphRagClient,
) -> Result<Option<String>, ContentAgentError> {
    // Map variable names to entity queries
    let entity_type = match var_name {
        "organization" => "Organization",
        "department" => "Department",
        "contact_person" => "Person",
        "location" => "Location",
        _ => return Ok(None),
    };
    
    // Query knowledge graph for entity
    let query = format!(
        "MATCH (e:{}) RETURN e.name LIMIT 1",
        entity_type
    );
    
    match kg_client.query(&query).await {
        Ok(results) => {
            if let Some(first) = results.first() {
                if let Some(name) = first.get("name") {
                    return Ok(Some(name.clone()));
                }
            }
            Ok(None)
        }
        Err(e) => Err(ContentAgentError::KnowledgeGraphError(
            e.to_string()
        )),
    }
}

/// Add conditional section flags to context
fn add_conditional_sections(
    context: &mut tera::Context,
    research: &ResearchContext,
    template: &Template,
) {
    // Set flags for conditional sections based on research context
    for section_name in &template.optional_sections {
        let should_include = research
            .suggested_sections
            .contains(section_name);
        context.insert(format!("include_{}", section_name), &should_include);
    }
}

/// Generate AI-powered content sections
async fn generate_ai_sections(
    context: &mut tera::Context,
    research: &ResearchContext,
    config: &ContentConfig,
) -> Result<(), ContentAgentError> {
    // Generate standard sections like summary, background, etc.
    let sections_to_generate = vec![
        ("introduction", generate_introduction),
        ("background", generate_background),
        ("conclusion", generate_conclusion),
    ];
    
    for (section_name, generator) in sections_to_generate {
        if !context.contains_key(section_name) {
            let content = generator(research, config).await?;
            context.insert(section_name, &content);
        }
    }
    
    Ok(())
}

    }
